18//! Doubly-buffered value with non-blocking read and blocking write.
19//!
20//! Logically, this is a single-value container, which has getter and setter
21//! with the following characteristics:
22//!
23//! - getters are running concurrently, and setters are serialized
24//!
25//! - setter is blocking; it may be blocked by both getters and setters,
26//! but in the average case only by setters
27//!
28//! - getter is non-blocking and lock-free; it does not block if a setter
29//! thread is suspended in the middle
30//!
31//! - getter is not wait-free though; if setters are called too frequently
32//! and have higher priority, then getter may spin until it has a chance
33//! to cut in between
34//!
35//! - sequential consistency is provided; after a setter returns, it's
36//! guaranteed that subsequent getters will observe up-to-date value
37//!
38//! Physically container is implemented as two copies of the value:
39//!
40//! - one copy is read-only and used by getters (concurrently)
41//!
42//! - another one is for setters; basically, the setter updates its copy
43//! while it's not visible to getters, waits until all getter using
44//! another copy finish, and switches the copies
45//!
46//! The algorithm is optimized for the following use case:
47//!
48//! - getters are frequent and setters are infrequent
49//!
50//! - the value is not very large and it's acceptable to make
51//! extra copies
52//!
53//! The value stored in the double buffer is immutable after it's set. To
54//! change the value, you need to call the getter, make a copy, modify it,
55//! and pass it to the setter.
56//!
57//! The value should have public default and copy constructors. If it also
58//! has move constructor, it can be used in setter, and then only one copy
59//! is needed to update value. Otherwise, two copies are needed.
60//!
61//! Typical reader looks like the following:
62//! @code
63//! DoubleBuffer<T> fooBuf;
64//! ...
65//! {
66//! auto readLock = fooBuf.GetReadLock();
67//!
68//! const T& fooValue = readLock.GetReference();
69//! // safe read-only access to fooValue until block end
70//! }
71//! @endcode
72//!
73//! Or, if the reader is okay to call the copy constructor:
74//! @code
75//! DoubleBuffer<T> fooBuf;
76//! ...
77//! T fooValue = fooBuf.Get();
78//! @endcode
79//!
80//! Typical writer looks like this:
81//! @code
82//! DoubleBuffer<T> fooBuf;
83//! ...
84//! T fooValue = fooBuf.Get();
85//! // modify fooValue
86//! fooBuf.Set(std::move(fooValue));
87//! @endcode
88//!
89//! @see ReadLock.
90template <typename T>
91class DoubleBuffer
92{
93 using BufferIndex = UInt64;
94
95 struct Buffer
96 {
97 T value;
98 std::atomic<BufferIndex> index = 0;
99 mutable std::shared_mutex mutex;
100 };
101
102public:
103 //! Read lock.
104 //!
105 //! Similar to scoped locks like std::lock_guard, but also provides
106 //! GetReference() method to access the value.
107 //!
108 //! While the lock is alive, it's safe to access the data for reading.
109 class ReadLock
110 {
111 public:
112 //! Acquire the lock.
113 //! Non-blocking and lock-free.
114 //! May spin for a while if a setter is running concurrently, but wont
115 //! block if the setter is suspended somewhere on the halfway.
116 ReadLock(const DoubleBuffer& doubleBuffer)
117 {
118 for (;;) {
119 // Load current buffer index.
120 const auto index = doubleBuffer.currentIndex_.load();
121
122 // Get reference to the current buffer.
123 auto& currentBuffer = doubleBuffer.GetBufferAt(index);
124
125 // Try to lock buffer for reading.
126 //
127 // The lock can fail in the case when, after we loaded the current
128 // index, but before obtained the lock, the setter was called and
129 // finished once, and then was called another time and is currently
130 // in progress. Here we should retry and re-read current index.
131 //
132 // We're still lock-free because even if the ongoing setter is
133 // suspended, we wont block. Instead, on the next try we will
134 // switch to another buffer and will successfully obtain the lock.
135 //
136 // However, we're not wait-free because while this situation is
137 // repeating and new setters continue to come and preempt us, we'll
138 // have to retry. Fortunately, this is very unlikely to happen.
139 if (!currentBuffer.mutex.try_lock_shared()) {
140 continue;
141 }
142
143 // Check that the buffer index still matches the our current index
144 // after obtaining the lock.
145 //
146 // This check catches the case similar to the above one, but when
147 // the second setter is already finished.
148 if (currentBuffer.index.load() != index) {
149 currentBuffer.mutex.unlock_shared();
150 continue;
151 }
152
153 // At this point, it is guaranteed that any setter will see that
154 // we're using the buffer and wont rewrite it until we release
155 // the lock in destructor.
156 buffer_ = &currentBuffer;
157 return;
158 }
159 }
160
161 //! Release the lock.
162 ~ReadLock()
163 {
164 if (buffer_) {
165 buffer_->mutex.unlock_shared();
166 }
167 }
168
169 //! Move lock.
170 ReadLock(ReadLock&& other)
171 {
172 std::swap(buffer_, other.buffer_);
173 }
174
175 ReadLock(const ReadLock&) = delete;
176 ReadLock& operator=(const ReadLock&) = delete;
177 ReadLock& operator=(ReadLock&&) = delete;
178
179 //! Get read-only reference to the value.
180 //! The reference may be used until read lock destructor is called.
181 //! The referred value is guaranteed to be immutable until that.
182 const T& GetReference() const
183 {
184 return buffer_->value;
185 }
186
187 private:
188 const Buffer* buffer_ = nullptr;
189 };
190

195 }

198 explicit DoubleBuffer(T&& value = T())

208 //! Get locked read-only reference to the value.
209 //! Non-blocking and lock-free, see ReadLock.
210 ReadLock GetReadLock() const
211 {
212 return ReadLock(*this);
213 }
214
215 //! Get copy of the value.
216 //! Non-blocking and lock-free, see ReadLock.
217 T Get() const
218 {
219 ReadLock readLock(*this);
220
221 return T(readLock.GetReference());
222 }

224 //! Set value.
225 //!
226 //! Blocks until all getters, invoked before this call, are finished.
227 //! Concurrent setter calls are serialized.
228 //!
229 //! It is guaranteed that if a getter is called after a setter returns,
230 //! the getter will observe the updated value.
231 //!
232 //! The value is either copied or moved into the internal buffer,
233 //! depending on whether an lvalue or rvalue is passed.
234 template <typename TT>
235 void Set(TT&& value)
236 {
237 // Serialize setters.
238 std::lock_guard<std::mutex> writeLock(writeMutex_);
239
240 // Load current buffer index.
241 // Since this index is modified only by setters, it's safe to use "relaxed".
242 const auto index = currentIndex_.load(std::memory_order_relaxed);
243
244 // Get reference to the "other" buffer, i.e. not the current one.
245 auto& otherBuffer = GetBufferAt(index + 1);
246
247 // Notify getters which were started earlier and are working or going to work
248 // with the previous buffer that this buffer is going to be invalidated.
249 otherBuffer.index = (index + 1);
250
251 // Wait until finishing of ongoing getters that are still using this buffer.
252 //
253 // After we obtain the lock, we can be sure that old getters are either finished
254 // or will see the index which we've updated above and wont use the buffer.
255 //
256 // Note that since it is the previous buffer, not the current one, chances are
257 // that there are no getters already, and most times we will acquire the lock
258 // immediately here.
259 std::unique_lock<decltype(otherBuffer.mutex)> lock(otherBuffer.mutex);
260
261 // Forward the value to the buffer.
262 otherBuffer.value = std::forward<TT>(value);
263
264 // Switch current buffer index to the other buffer.
265 // It is guaranteed that all getters invoked after this point will return
266 // the new value.
267 currentIndex_ = (index + 1);
268 }
269
270private:
271 friend class ReadLock;
272
273 const Buffer& GetBufferAt(BufferIndex index) const
274 {
275 return buffers_[index % 2];
276 }
277
278 Buffer& GetBufferAt(BufferIndex index)
279 {
280 return buffers_[index % 2];
281 }
282
283 std::mutex writeMutex_;
284
285 Buffer buffers_[2];
286 std::atomic<BufferIndex> currentIndex_ = 0;
287};
